-- SQLite Database Schema --

Tables and Columns:
AUTHOR(author_id, name, surname, nationality, birth_date, death_date)
EMPLOYEE(employee_id, name, surname, role, hire_date, email, phone)
ROOM(room_id, responsible_employee_id, name, capacity, floor, type)
READER(reader_id, name, surname, email, phone, address, subscription_date)
BOOK(book_id, author_id, title, publication_year, genre, num_copies)
BOOK_LOAN(book_id, reader_id, loan_date, return_date, notes)
ROOM_RESERVATION(reader_id, date, start_time, end_time, room_id, notes)

Primary Keys:
AUTHOR(author_id)
EMPLOYEE(employee_id)
ROOM(room_id)
READER(reader_id)
BOOK(book_id)
BOOK_LOAN(book_id, reader_id, loan_date)
ROOM_RESERVATION(reader_id, date, start_time)

Foreign Key Relationships:
BOOK.author_id -> AUTHOR.author_id
ROOM.responsible_employee_id -> EMPLOYEE.employee_id
BOOK_LOAN.book_id -> BOOK.book_id
BOOK_LOAN.reader_id -> READER.reader_id
ROOM_RESERVATION.reader_id -> READER.reader_id
ROOM_RESERVATION.room_id -> ROOM.room_id
